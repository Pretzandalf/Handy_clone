    &F_{gravity} = mg& \\
    &F_{resistance} = \alpha {\upsilon}^2 & \\
    &F_{Magnus} = \beta \begin{bmatrix} \vec \omega \times \vec v \end{bmatrix} & 
\end{flalign*}

\begin{flalign*}
&
m \begin{bmatrix}
\ddot x \\
\ddot y \\
\ddot z \\
\end{bmatrix}
= - \alpha \begin{bmatrix}
{\dot x}^2 \\
{\dot y}^2 \\
{\dot z}^2 \\
\end{bmatrix} -\beta \begin{bmatrix}
\vec \omega \times \vec v \\
\end{bmatrix} - m \begin{bmatrix}
0 \\
0 \\
g \\
\end{bmatrix}& \\
&
\begin{bmatrix}
\vec \omega \times \vec v \\
\end{bmatrix} 
= \det \begin{bmatrix}
\vec i & \vec j & \vec k \\
\omega_x & \omega_y &\omega_z \\
\dot x & \dot y & \dot z \\
\end{bmatrix} = \begin{bmatrix}
0 \cdot \dot x - \omega_z \dot y +\omega_y \dot z \\
\omega_z \dot x + 0 \cdot \dot y  - \omega_x \dot z \\
- \omega_y \dot x + \omega_x \dot y + 0\cdot \dot z \\
\end{bmatrix} & 
\end{flalign*}

\begin{flalign*}
&
m \begin{bmatrix}
\ddot x \\

\ddot z \\

= - \begin{bmatrix}
\alpha {\dot x}^2 - \beta \omega_z \dot y + \beta \omega_y \dot z \\
\beta \omega_z \dot x + \alpha {\dot y}^2  - \beta \omega_x \dot z \\
- \beta \omega_y \dot x + \beta \omega_x \dot y + \alpha {\dot z}^2 \\
\end{bmatrix} - m \begin{bmatrix}
0 \\
0 \\
g \\
\end{bmatrix}
&
\end{flalign*}
